        // Function to create a dog card

        function createDogCard(dog) {

            const card = document.createElement('div');

            card.className = 'bg-white rounded-xl shadow-lg hover:shadow-xl transition-shadow duration-300 overflow-hidden flex flex-col';

            card.innerHTML = `

                <img src="${dog.image}" alt="${dog.name}" class="card-image rounded-t-xl">

                <div class="p-6 flex flex-col flex-grow">

                    <h3 class="text-2xl font-bold mb-2 text-gray-800">${dog.name}</h3>

                    <p class="text-gray-600 mb-4 flex-grow">${dog.shortDescription}</p>

                    <button data-dog-id="${dog.id}" class="learn-more-button bg-blue-500 hover:bg-blue-600 text-white font-bold py-3 px-6 rounded-lg shadow-md transition duration-300 ease-in-out transform hover:scale-105 self-end">了解更多</button>

                </div>

            `;

            return card;

        }

 

        // Function to render dog cards

        function renderDogCards(breeds) {

            dogCardsGrid.innerHTML = ''; // Clear existing cards

            if (breeds.length === 0) {

                dogCardsGrid.innerHTML = '<p class="text-center text-gray-600 col-span-full">没有找到匹配的狗狗品种。</p>';

                return;

            }

            breeds.forEach(dog => {

                dogCardsGrid.appendChild(createDogCard(dog));

            });

            addLearnMoreListeners(); // Add listeners after rendering

        }

 

        // Function to add event listeners to "Learn More" buttons

        function addLearnMoreListeners() {

            document.querySelectorAll('.learn-more-button').forEach(button => {

                button.addEventListener('click', (event) => {

                    const dogId = event.target.dataset.dogId;

                    const dog = dogBreeds.find(d => d.id === dogId);

                    if (dog) {

                        modalDogName.textContent = dog.name;

                        modalDogImage.src = dog.image;

                        modalDogIntroduction.textContent = dog.introduction;

                        modalDogFeeding.textContent = dog.feeding;

                        modalDogPersonality.textContent = dog.personality;

                        dogDetailModal.classList.remove('hidden');

                        // Add animation classes

                        dogDetailModal.querySelector('div').classList.remove('scale-95', 'opacity-0');

                        dogDetailModal.querySelector('div').classList.add('scale-100', 'opacity-100');

                    }

                });

            });

        }

 

        // Close modal event listener

        closeModalButton.addEventListener('click', () => {

            // Remove animation classes before hiding

            dogDetailModal.querySelector('div').classList.remove('scale-100', 'opacity-100');

            dogDetailModal.querySelector('div').classList.add('scale-95', 'opacity-0');

            setTimeout(() => {

                dogDetailModal.classList.add('hidden');

            }, 300); // Match transition duration

        });

 

        // Close modal when clicking outside

        dogDetailModal.addEventListener('click', (event) => {

            if (event.target === dogDetailModal) {

                // Remove animation classes before hiding

                dogDetailModal.querySelector('div').classList.remove('scale-100', 'opacity-100');

                dogDetailModal.querySelector('div').classList.add('scale-95', 'opacity-0');

                setTimeout(() => {

                    dogDetailModal.classList.add('hidden');

                }, 300); // Match transition duration

            }

        });

 

        // Search functionality

        searchButton.addEventListener('click', () => {

            const searchTerm = searchInput.value.toLowerCase().trim();

            const filteredBreeds = dogBreeds.filter(dog =>

                dog.name.toLowerCase().includes(searchTerm) ||

                dog.shortDescription.toLowerCase().includes(searchTerm) ||

                dog.introduction.toLowerCase().includes(searchTerm) ||

                dog.feeding.toLowerCase().includes(searchTerm) ||

                dog.personality.toLowerCase().includes(searchTerm)

            );

            renderDogCards(filteredBreeds);

        });

 

        // Initial render of all dog cards

        document.addEventListener('DOMContentLoaded', () => {

            renderDogCards(dogBreeds);
